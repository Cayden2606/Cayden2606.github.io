<script>
  function openLightbox(imgEl) {
    const src = imgEl.getAttribute('src') || '';
    const fullSrc = src.replace('/compressed/', '/').replace('-compressed', ''); // map to full-size
    const lb = document.getElementById('lightbox');
    const lbImg = document.getElementById('lightboxImg');

    lbImg.src = fullSrc;
    lbImg.alt = imgEl.alt || '';
    lb.style.display = 'flex';

    // Prevent background scroll
    document.body.style.overflow = 'hidden';
  }

  function closeLightbox(e) {
    // Close when clicking overlay, or clicking the close button
    const lb = document.getElementById('lightbox');
    const lbImg = document.getElementById('lightboxImg');

    // If click is on the image itself, do nothing
    if (e && e.target === lbImg) return;

    lb.style.display = 'none';
    lbImg.src = '';
    document.body.style.overflow = '';
  }

  document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') {
      const lb = document.getElementById('lightbox');
      if (lb && lb.style.display === 'flex') closeLightbox(e);
    }
  });
</script>
